Dịch Vụ JWT Tự Chế (Self-contained JWT)

Hệ thống tự thực hiện quá trình mã hóa/giải mã JWT bằng HMAC-SHA256, không dựa vào thư viện bên thứ ba giúp bảo mật tối ưu và kiểm soát 100% dòng đời Token.

  • Không phụ thuộc Package ngoài (Zero dependency JWT).
  • Cơ chế Token Blacklisting bảo mật qua bảng tokens.
  • Hỗ trợ Access Token ngắn hạn & Refresh Token dài hạn.

// routes/api.php - Định tuyến chuẩn RESTful API với các lớp phân quyền và JWT
use App\Http\Controllers\Api\V1\ProductController;
use Illuminate\Support\Facades\Route;

Route::prefix('v1')->group(function () {
    
    Route::prefix('products')->group(function () {
        // Quản lý tồn kho & sản phẩm - Dành riêng cho ADMIN, STAFF
        Route::get('management', [ProductController::class, 'management'])
            ->middleware(['jwt', 'role:ADMIN,STAFF']);

        // APIs công khai (Public)
        Route::get('', [ProductController::class, 'index']);
        Route::get('{product}', [ProductController::class, 'show']);
        
        // APIs chỉnh sửa sản phẩm - Dành riêng cho ADMIN
        Route::post('', [ProductController::class, 'store'])
            ->middleware(['jwt', 'role:ADMIN']);
        Route::put('{product}', [ProductController::class, 'update'])
            ->middleware(['jwt', 'role:ADMIN']);
    });
});
<?php

// app/Http/Controllers/Api/V1/ProductController.php - Skinny Controller điều phối
namespace App\Http\Controllers\Api\V1;

use App\Http\Requests\Products\ProductRequest;
use App\Http\Resources\ProductResource;
use App\Services\ProductService;
use App\Services\ApiResponseService;
use Illuminate\Http\JsonResponse;

class ProductController
{
    // Dependency Injection các Service liên quan vào constructor
    public function __construct(
        private ApiResponseService $response, 
        private ProductService $productService
    ) {}

    public function store(ProductRequest $request): JsonResponse
    {
        // Validate đã tự động thực thi bởi Form Request, chỉ gọi Service xử lý nghiệp vụ
        $product = $this->productService->create($request->validated());

        return $this->response->success(
            'Tạo sản phẩm thành công', 
            (new ProductResource($product))->toArray($request), 
            201
        );
    }
}
<?php

// app/Services/ProductService.php - Chứa Business Logic nghiệp vụ
namespace App\Services;

use App\Models\Product;
use Illuminate\Support\Facades\DB;

class ProductService
{
    public function create(array $data): Product
    {
        // Đảm bảo an toàn giao dịch database khi lưu thông tin và thuộc tính động JSON
        return DB::transaction(function () use ($data) {
            $product = Product::create([
                'name' => $data['name'],
                'price' => $data['price'],
                'quantity' => $data['quantity'],
                'category_id' => $data['category_id'],
                'specifications' => $data['specifications'] ?? null,
                'attributes' => $data['attributes'] ?? null,
            ]);

            // Tự động cập nhật lịch sử Log tồn kho ban đầu
            $product->inventoryLogs()->create([
                'change_type' => 'IN',
                'quantity_change' => $data['quantity'],
                'reason' => 'Khởi tạo sản phẩm mới',
                'performed_by' => auth()->id(),
            ]);

            return $product;
        });
    }
}
<?php

// app/Services/JwtService.php - Dịch vụ JWT tự chế bảo mật
namespace App\Services;

use App\Models\User;

class JwtService
{
    public function createAccessToken(User $user): string
    {
        return $this->encode([
            'sub' => $user->username,
            'user_id' => $user->id,
            'role' => $user->role->name,
            'exp' => time() + config('shop.jwt_expiration_seconds'),
            'type' => 'access',
        ]);
    }

    private function encode(array $payload): string
    {
        $header = [
            'typ' => 'JWT',
            'alg' => config('shop.jwt_algorithm'),
        ];
        $encodedHeader = $this->base64UrlEncode(json_encode($header));
        $encodedPayload = $this->base64UrlEncode(json_encode($payload));
        
        // Ký mã xác thực signature bằng HMAC-SHA256 bảo mật
        $signature = $this->sign("{$encodedHeader}.{$encodedPayload}");

        return "{$encodedHeader}.{$encodedPayload}.{$signature}";
    }

    private function sign(string $value): string
    {
        return $this->base64UrlEncode(
            hash_hmac('sha256', $value, config('shop.jwt_secret'), true)
        );
    }
}
